import math

def sieve_of_eratosthenes(n):
    x = [True] * n
    x[0] = x[1] = False

    for j in range(2, math.isqrt(n) + 1):
        if x[j]:
            for i in range(j*j, n, j):
                x[i] = False

    prime_indices = [i for i in range(n) if x[i]]
    return prime_indices
